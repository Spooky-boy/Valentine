    <script>
        const yesBtn = document.getElementById('yesBtn');
        const noBtn = document.getElementById('noBtn');
        const askSection = document.getElementById('askSection');
        const message = document.getElementById('message');
        const questionText = document.getElementById('questionText');

        let yesFontSize = 18;
        let clickCount = 0;

        const dynamicTexts = [
            "Yakin?",
            "Beneran nih?",
            "Pikir-pikir dulu...",
            "Yah, kok gitu?",
            "Pliss donggg",
            "Nanti nyesel lho!",
            "Satu kesempatan lagi?",
            "Jahat banget sih :("
        ];

        function moveButton() {
            // Membuat tombol "Enggak" berpindah tempat
            const x = Math.random() * (window.innerWidth - noBtn.offsetWidth);
            const y = Math.random() * (window.innerHeight - noBtn.offsetHeight);
            
            noBtn.style.position = 'fixed';
            noBtn.style.left = `${x}px`;
            noBtn.style.top = `${y}px`;

            // Mengubah teks pertanyaan dan memperbesar tombol "Mau"
            if (clickCount < dynamicTexts.length) {
                noBtn.innerText = dynamicTexts[clickCount];
            }
            
            clickCount++;
            yesFontSize += 10;
            yesBtn.style.fontSize = `${yesFontSize}px`;
            yesBtn.style.padding = `${yesFontSize/2}px ${yesFontSize}px`;
        }

        noBtn.addEventListener('mouseover', moveButton);
        noBtn.addEventListener('click', moveButton);

        yesBtn.addEventListener('click', () => {
            askSection.style.display = 'none';
            message.style.display = 'block';
        });
